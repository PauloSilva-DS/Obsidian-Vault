```python
def fibonacci(n):
    if n <= 1:
        return n
    else:
        return fibonacci(n-1) + fibonacci(n-2)

# Test the function with some inputs
print(fibonacci(5)) # Output: 5
print(fibonacci(9)) # Output: 21
```
